    回文自动机

    简介¶

    回文自动机这东西还是比较新的.
    回文自动机是一棵|\Sigma|(字符集大小)叉树,用ch[x][i]表示x的第i个儿子.
    回文自动机的每个节点都代表一个回文串,其第i个儿子的回文串为该节点的回文串在两边各加上一个字符i.
    同时还有fail[i]表示节点i的回文串最长的回文后缀对应的节点.
    很明显一个回文串的回文后缀同时也是回文前缀,因此在当前节点时,最长回文后缀已经有对应的节点.
    为了方便阐述,以下提到某节点时也有"该节点所表示的字符串"的含义,反之也是如此. 不然好难讲的(*/ω＼*))

    过程¶

    怎么构造一棵回文自动机呢?
    设当前做到第i个字符,当前p指向以第i-1个字符结尾的最长回文串(设为q).
    若第i个字符与q第一个字符之前的字符相等(称为匹配成功),直接将这第i个字符加到q的两边,作为q的子节点.
    否则的话q=fail[q],重复如上过程直至匹配成功.
    当然,匹配后的字符串(p赋为该字符串)有可能是一个之前没有出现过的节点,此时就需要新建节点.
    为了求fail[p],可以像上一步那样由fail[q]不断匹配得到.
    回文自动机应该不难理解.一些边界如何处理可以参考代码.不理解可以对照样例和代码画图.

    #include<bits/stdc++.h>
#define u32 unsigned
#define i64 long long
#define u64 unsigned long long
#define f80 long double
#define rgt register
using namespace std;
#define MAXN 300005

char s[MAXN];
int ch[MAXN][26], fail[MAXN], len[MAXN], cnt[MAXN], tot(-1), p, q;
i64 ans;

inline int newnode( int x ){ return len[++tot] = x, tot; }
inline int getfail( int x, int n ){ while( s[n] != s[n - len[x] - 1] ) x = fail[x]; return x; }

signed main(){
    scanf( "%s", s + 1 );
    newnode(0), newnode(-1), fail[0] = p = 1, s[0] = 96;
    //一些边界处理细节
    for ( int i = 1; s[i] != '\0'; ++i ){
        s[i] -= 'a', q = getfail(p, i); //匹配
        if ( !ch[q][s[i]] ){
            p = newnode(len[q] + 2), //新建节点
            fail[p] = ch[getfail(fail[q], i)][s[i]], //匹配
            ch[q][s[i]] = p; //再连边,必须在上一步之后
        } ++cnt[p = ch[q][s[i]]]; //计数
    } for ( int i = tot; i; --i ) ans = max( ans, 1ll * len[i] * cnt[i] ), cnt[fail[i]] += cnt[i];
    printf( "%lld\n", ans );
    return 0;
}

    洛谷 P4287 [SHOI2011]双倍回文¶

    可以记录一个数组tr[i]表示最长的,不超过i长度一半的回文后缀.

    #include<bits/stdc++.h>
using namespace std;
#define MAXN 500005

int N, ch[MAXN][26], len[MAXN], fail[MAXN], tr[MAXN], tot(-1);
char s[MAXN];
inline int newnode( int x ){ return len[++tot] = x, tot; }
inline int get( int x, int n ){
    while( s[n] != s[n - len[x] - 1] ) x = fail[x];
    return x;
}

int main(){
    scanf( "%d", &N ), scanf( "%s", s + 1 ), s[0] = 26;
    newnode(0), newnode(-1), fail[0] = 1;
    for ( int i = 1, p = 1, q; i <= N; ++i ){
        s[i] -= 'a', q = get(p, i);
        if ( !ch[q][s[i]] ){
            fail[p = newnode(len[q] + 2)] = ch[get(fail[q], i)][s[i]], ch[q][s[i]] = p;
            if ( len[p] <= 2 ) tr[p] = fail[p];
            else{ int t(tr[q]);
                while( ( ( len[t] + 2 ) << 1 ) > len[p] || s[i] != s[i - len[t] - 1] ) t = fail[t];
                tr[p] = ch[t][s[i]];
            }
        } p = ch[q][s[i]];
    } int ans(0);
    for ( int i = 0; i <= tot; ++i ) if ( !( len[i] & 3 ) && ( len[tr[i]] << 1 ) == len[i] ) ans = max( ans, len[i] );
    printf( "%d\n", ans ); 
    return 0;
}
