    二维凸包

    简介¶

    在一个平面内有一些点
    所谓二维凸包（以下简称凸包），通俗地讲，就是将一条橡皮筋圈住所有的点，放手后橡皮筋围成的图形。
    容易发现，凸包也是能圈住所有点的最小多边形（不仅面积最小，周长也是最小的）。

    求法¶

    凸包有Graham扫描法和Andrew算法两种做法。
    这里介绍比较好写的Andrew算法。
    首先，对所有的点排序（横坐标为第一关键字，纵坐标为第二关键字，升序）
    容易发现，最小的点S（即横坐标最小的前提下纵坐标最小）和最大的点T（同理）肯定是凸包上的点。
    事实上，只要横坐标最小或横坐标最大或纵坐标最小或纵坐标最大，这个点就一定是凸包上的点，但是没必要全部预处理出来。
    我们将这个凸包从ST一分为二，对上下两部分分别求解。
    先从S开始，按序扫描每一个点。
    先找上半个凸包，很明显斜率应该越来越小。
    维护一个单调栈（确切地说，单调栈里面存的应该是边，边的斜率递减），S先入栈（还没有一条边）
    每遇到一个点，先吧这个点与栈顶连一条边（这是想象操作，代码实现中并不需要真的连起来），把之前斜率小于（等于）这条边的统统弹出，然后再把这个点push进栈（实际上就是刚才连的那条边）。
    当扫完所有点时，再倒过来扫一遍就OK了。
    实际操作中，为了方便，避免除法，可以使用叉积来优化。

    #include<bits/stdc++.h>
using namespace std;
#define MAXN 10005
#define rgt register

struct vec{
    double x, y; vec( double a = 0., double b = 0. ):x(a), y(b){}
    inline bool operator < ( const vec &a ){ return x == a.x ? y < a.y : x < a.x; } 
    inline double operator * ( const vec &a ){ return x * a.y - a.x * y; } //叉积
    inline vec operator - ( const vec &a ){ return vec( x - a.x, y - a.y ); }
    inline vec operator + ( const vec &a ){ return vec( x + a.x, y + a.y ); }
    inline double length(){  return sqrt( x * x + y * y ); }
}v[MAXN];

int N, s[MAXN], tp; double ans;

int main(){
    scanf( "%d", &N );
    for ( rgt int i = 1; i <= N; ++i ) scanf( "%lf%lf", &v[i].x, &v[i].y );
    sort( v + 1, v + N + 1 ), s[tp = 1] = 1;
    for ( rgt int i = 2; i <= N; s[++tp] = i, ++i )
        while( tp > 1 && ( v[s[tp]] - v[s[tp - 1]] ) * ( v[i] - v[s[tp]] ) > 0 ) --tp; //不在凸包上,弹出
    for ( rgt int i = N - 1, t(tp); i; s[++tp] = i, --i )
        while( tp > t && ( v[s[tp]] - v[s[tp - 1]] ) * ( v[i] - v[s[tp]] ) > 0 ) --tp; 
    for ( rgt int i = 1; i < tp; ++i ) ans += ( v[s[i + 1]] - v[s[i]] ).length();//最后求出周长
    printf( "%.2lf\n", ans ); return 0;
}

#define u32 unsigned
#define i64 long long
#define u64 unsigned long long
#define f80 long double
#define rgt register
#define getchar() ( p1 == p2 && ( p2 = bf + fread( bf, 1, 1 << 21, stdin ), p1 = bf ) == p2 ? EOF : *p1++ )
using namespace std;
#define MAXN 100005

char bf[1 << 21], *p1, *p2;
template<typename T>
inline void read( rgt T &x ){ x = 0; rgt char t, flg(0);
    for ( t = getchar(); !isdigit(t); t = getchar() ) flg = t == '-';
    for ( ; isdigit(t); t = getchar() ) x = x * 10 + ( t & 15 ); x = flg ? -x : x;
}

struct vec{
    int x, y; vec( int a = 0, int b = 0 ):x(a), y(b){}
    inline bool operator < ( const vec &t )const{ return x == t.x ? y < t.y : x < t.x; }
    inline vec operator - ( const vec &t ){ return vec( x - t.x, y - t.y ); }
    inline i64 operator * ( const vec &t ){ return (i64)x * t.y - (i64)y * t.x; }
}v[MAXN];

int T, N, s[MAXN], tp;

signed main(){
    read(T);
    while( T-- ){
        read(N);
        for ( int i = 1; i <= N; ++i ) read(v[i].x), read(v[i].y), getchar() == 'N' ? --i, --N : 0;
        sort( v + 1, v + N + 1 ), s[tp = 1] = 1;
        for ( int i = 2; i <= N; s[++tp] = i, ++i ) while( tp > 1 && ( v[s[tp]] - v[s[tp - 1]] ) * ( v[i] - v[s[tp]] ) < 0 ) --tp;
        for ( int i = N - 1, t(tp); i; s[++tp] = i, --i ) while( tp > t && ( v[s[tp]] - v[s[tp - 1]] ) * ( v[i] - v[s[tp]] ) < 0 ) --tp;
        printf( "%d\n", tp - 1 ); for ( int i = 1; i < tp; ++i ) printf( "%d %d\n", v[s[i]].x, v[s[i]].y );
    }
    return 0;
}

    洛谷P3829 [SHOI2012]信用卡凸包¶

    信用卡凸包可以分为线段部分和曲线部分（曲线部分肯定是四角圆的一部分），把线段部分统统往里面移，使端点落在四角的圆心上，构成的图形就是将所有四角圆心取出后的凸包。然后求个凸包就可以完美解决。别忘了还有曲线部分，很明显，这些曲线能拼成一整个圆（简单多边形外角和为2\pi），比线段部分还好求。

    #include<bits/stdc++.h>
#define u32 unsigned
#define i64 long long
#define u64 unsigned long long
#define f80 long double
#define rgt register
using namespace std;
#define MAXN 100005
#define pi 3.141592653589793
#define eps 1e-12

struct vec{
    double x, y; vec( double a = 0., double b = 0. ):x(a), y(b){}
    inline double operator * ( const vec &a ){ return x * a.y - a.x * y; }
    inline bool operator < ( const vec &a ){ return x == a.x ? y < a.y : x < a.x; }
    inline vec operator - ( const vec &a ){ return vec( x - a.x, y - a.y ); }
    inline vec operator + ( const vec &a ){ return vec( x + a.x, y + a.y ); }
    inline vec rotate( double dg ){ static double s, c; s = sin(dg), c = cos(dg); return vec( x * c - y * s, x * s + y * c ); } //向量旋转
    inline double length(){  return sqrt( x * x + y * y ); }
}v[MAXN << 2], t1, t2, t3, t4, o;

int n, N, stk[MAXN << 2], tp;
double a, b, r, x, y, th;

signed main(){
    scanf( "%d%lf%lf%lf", &n, &a, &b, &r ), a /= 2., b /= 2.,
    t1 = vec(b - r, a - r), t2 = vec(b - r, r - a),
    t3 = vec(r - b, a - r), t4 = vec(r - b, r - a);
    for ( int i = 1; i <= n; ++i )
        scanf( "%lf%lf%lf", &x, &y, &th ), o = vec( x, y ), x += eps, y += eps, th += eps,
        v[++N] = o + t1.rotate(th), v[++N] = o + t2.rotate(th),
        v[++N] = o + t3.rotate(th), v[++N] = o + t4.rotate(th);
    sort( v + 1, v + N + 1 ), stk[tp = 1] = 1;
    for ( int i = 2; i <= N; stk[++tp] = i, ++i )
        while( tp > 1 && ( v[stk[tp]] - v[stk[tp - 1]] ) * ( v[i] - v[stk[tp]] ) <= 0 ) --tp;
    for ( int i = N - 1, tmp(tp); i; stk[++tp] = i, --i )
        while( tp > tmp && ( v[stk[tp]] - v[stk[tp - 1]] ) * ( v[i] - v[stk[tp]] ) <= 0 ) --tp;
    double ans(2 * r * pi);
    for ( int i = 1; i < tp; ++i ) ans += ( v[stk[i]] - v[stk[i + 1]] ).length();
    return printf( "%.2lf\n", ans ), 0;
}
