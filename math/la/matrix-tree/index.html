    矩阵树定理¶

    先讲讲这是干嘛用的。 矩阵树定理用于求有向图/无向图的生成树个数。

    这种这么难的东西我不会证明啊qwq 滚去学学线性代数再来填坑好了qwq 填坑时间的话0~+\infty年不等

    首先构造邻接矩阵A以及度数矩阵D，基尔霍夫矩阵(Kirchhoff)为K=D-A。 讲的通俗一点，把加边操作变成这样： 

    1
2
3
    inline void add( int x, int y ){
    ++a[x][x], ++a[y][y], --a[x][y], --a[y][x];
}
    a即为基尔霍夫矩阵。 这是无向图的加边方式。 对于有向图的情况，如果要求是外向树(所有边方向都远离根节点)，建边方式为

    1
2
3
    inline void add( int x, int y ){
    ++a[y][y], --a[x][y];
}

    如果要求是内向树，建边方式为

    1
2
3
    inline void add( int x, int y ){
    ++a[x][x], --a[x][y];
}

    然后对根节点求余子式(也就是去掉根节点求行列式)即可。

    #include<bits/stdc++.h>
using namespace std;
#define Re register
#define MAXN 20
#define LD long double

int T, N, M, x, y;
LD a[MAXN][MAXN];

int main(){
    scanf( "%d", &T );
    while( T-- ){
        scanf( "%d%d", &N, &M ); N--;
        memset( a, 0, sizeof a );
        for ( int i = 1; i <= M; ++i ) scanf( "%d%d", &x, &y ), a[x][x] += 1, a[y][y] += 1, a[x][y] -= 1, a[y][x] -= 1;

        bool flg(1);

        for ( int i = 1; i <= N; ++i ){
            int mx(i);
            for ( int j = i + 1; j <= N; ++j ) if ( a[mx][i] < a[j][i] ) mx = j;

            if ( i != mx ) for ( int j = i; j <= N; ++j ) swap( a[i][j], a[mx][j] );

            if ( a[i][i] < 1e-8 && a[i][i] > -1e-8 ){ printf( "0\n" ); flg = 0; break; }

            for ( int j = i + 1; j <= N; ++j ){
                LD t(a[j][i] / a[i][i]);
                for ( int k = i; k <= N; ++k ) a[j][k] -= a[i][k] * t;
            }
        }
        if ( flg ){
            LD ans(1);
            for ( int i = 1; i <= N; ++i ) ans *= a[i][i];
            printf( "%.0Lf\n", ans >= 0 ? ans : -ans );
        }
    }
    return 0;
}

    #include<bits/stdc++.h>
using namespace std;
#define Re register
#define MAXN 99
#define LL long long
#define mod 1000000000

int N, M;
char s[13][13], cnt, c[13][13];
LL a[MAXN][MAXN];
bool flg;

void Add( int x, int y ){ a[x][y]--; a[y][x]--; a[x][x]++; a[y][y]++; }

int main(){
    scanf( "%d%d", &N, &M );
    for ( int i = 1; i <= N; ++i ) scanf( "%s", s[i] + 1 );
    for ( int i = 1; i <= N; ++i )
        for ( int j = 1; j <= M; ++j )
            if ( s[i][j] == '.' ) c[i][j] = ++cnt;
    for ( int i = 1; i <= N; ++i )
        for ( int j = 1; j <= M; ++j ){
            if ( !c[i][j] ) continue;
            if ( c[i][j + 1] ) Add( c[i][j], c[i][j + 1] );
            if ( c[i + 1][j] ) Add( c[i][j], c[i + 1][j] );
        }
    cnt--;

    flg = 1;

    for ( int i = 1; i <= cnt; ++i ){
        int mx(i);
        for ( int j = i + 1; j <= cnt; ++j ) if ( c[i][j] > c[i][mx] ) mx = i;
        if ( mx != i ){ flg = !flg; for ( int j = i; j <= cnt; ++j ) swap( c[i][j], c[i][mx] ); }
        if ( !a[i][i] ){ printf( "0\n" ); return 0; }
        for ( int j = i + 1; j <= cnt; ++j ){
            while( 1 ){
                int t(a[j][i] / a[i][i]);
                for ( int k = i; k <= cnt; ++k ) a[j][k] = ( ( a[j][k] - ( a[i][k] * t ) % mod ) % mod + mod ) % mod;
                if ( a[j][i] == 0 ) break;
                for ( int k = i; k <= cnt; ++k ) swap( a[i][k], a[j][k] );
                flg = !flg;
            }
        }
    }
    LL ans(1);
    for ( int i = 1; i <= cnt; ++i ) ans = ( ans * a[i][i] ) % mod;
    printf( "%lld\n", flg ? ans : mod - ans );
    return 0;
}
