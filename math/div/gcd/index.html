    #include<bits/stdc++.h>
using namespace std;

int gcd( int x, int y ){
    if ( y == 0 ) return x;
    return gcd( y, x % y );
}

int N, x, ans;

int main(){
    scanf( "%d", &N );
    for ( int i = 1; i <= N; ++i ){
        scanf( "%d", &x );
        if ( ans ) ans = gcd( ans, x );
        else ans = x;
        if ( ans < 0 ) ans = -ans;
    }
    printf( "%d\n", ans ); 
}

    欧几里得算法¶

    欧几里得算法用于求最大公约数。

    \forall \gcd( a, b ) = c, a > b,\text{有} \gcd( a - b, b ) = c

    证明:

    \because \gcd( a, b ) = c \\ \therefore c | a,c | b \\ \therefore c | ( a - b ), c | b \\ \therefore c | gcd( a - b, b ) \\ \text{此时我们已经完成一半了,接下来我们要证明不存在}(a-b)\text{和}b\text{之间没有更大的公约数} \\ \text{假设}d > c, d | ( a - b ), d | b \\ \therefore d | a \\ \therefore \gcd( a, b ) = d > c \therefore \text{假设不成立,原命题成立}

    然后我们就可以写代码辣qwq

    但是你会发现,更相减损术的复杂度不太对,一个gcd( 1, 1e9 )就能卡得TLE。

    扩展欧几里得算法¶

    根据贝祖定理，\forall a,b \in \mathbb{N^+}, \exists x, y,ax+by=\gcd(a,b) 扩展欧几里得算法就是用来求其中的x和y的。

    1
    void exgcd( int a, int b, int &x, int &y, int &z );

    其中z表示\gcd(a,b)
    很明显，当b=0时，解得x=1,y=0。
    否则，我们可以先解bx'+(a\%b)y'=\gcd(b,a\%b)=\gcd(a,b)
    转换一下可以变成bx'+(a\%b)y'=bx'+(a-\lfloor a/b\rfloor\times b)y'=ay'+b(x'-\lfloor a/b\rfloor\times y')
    于是此步解为x=y',y=x'-\lfloor a/b\rfloor\times y'

    void exgcd( int a, int b, int &x, int &y, int &z ){
    if ( !b ) return x = 1, y = 0, z = a, void();
    exgcd( b, a % b, y, x, z ), y -= a / b * x;
}
